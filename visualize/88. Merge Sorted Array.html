Mã nguồn thực thi (JS) merge-sorted.js
var merge = function(nums1, m, nums2, n) {
    let j = 0;
    let newM = m;
    for(let i = 0; i < nums1.length && j < n; i++) {
        
        // Gặp số 0 nằm ngoài ranh giới vùng chứa phần tử gốc
        if(nums1[i] === 0 && i >= newM) {
            nums1[i] = nums2[j];
            j++;
            continue;
        }
        
        // Trùng khớp: Chèn, dịch mảng, tăng cả 2 con trỏ
        if(nums1[i] === nums2[j]) {
            insert(nums1, i, nums2[j]);
            i++;
            j++;
            newM++;
        }
        // Phần tử nums1[i] lớn hơn: Chèn nums2[j] và dịch mảng
        else if(nums1[i] > nums2[j]) {
            insert(nums1, i, nums2[j]);
            j++;
            newM++;
        }
    }
};
 
// Hàm helper dịch chuyển các phần tử từ index về bên phải
var insert = function(array, index, value) {
    for(let i = array.length - 1; i >= index; i--) {
        array[i] = i === index ? value : array[i-1];
    }
}
